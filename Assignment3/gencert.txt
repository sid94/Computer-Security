# Have openssl installed in your work station (pip install pyopenssl).

# Run this below command in the dir where you want the private key and certificate to be genrated for the server use.

# Tip: if not working using the windows cmd prompt use git bash terminal

# Added all the interative command in the command itself or also a hack in -subj "//C=US/C=US....." this show a command req: Skipping unknown attribute "/C" but it will skip the first one 

# Command

openssl \
  req \
  -newkey rsa:4096 -nodes \
  -keyout privkey.pem \
  -x509 -days 365 -out skolhap1.pem \
  -subj "//C=US/C=US/ST=NY/L=Binghamton/O=siddhesh/OU=IT/CN=www.skolhap1.com/emailAddress=skolhap1@binghamton.edu"
